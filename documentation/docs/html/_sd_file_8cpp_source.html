34// add a cluster to a file
35uint8_t SdFile::addCluster() {
36 if (!vol_->allocContiguous(1, &curCluster_)) {
37 return false;
38 }
39
40 // if first cluster of file link to directory entry
41 if (firstCluster_ == 0) {
42 firstCluster_ = curCluster_;
43 flags_ |= F_FILE_DIR_DIRTY;
44 }
45 flags_ |= F_FILE_CLUSTER_ADDED;
46 return true;
47}
48//------------------------------------------------------------------------------
49// Add a cluster to a directory file and zero the cluster.
50// return with first block of cluster in the cache
51uint8_t SdFile::addDirCluster(void) {
52 if (!addCluster()) {
53 return false;
54 }
55
56 // zero data in cluster insure first cluster is in cache
57 uint32_t block = vol_->clusterStartBlock(curCluster_);
58 for (uint8_t i = vol_->blocksPerCluster_; i != 0; i--) {
59 if (!SdVolume::cacheZeroBlock(block + i - 1)) {
60 return false;
61 }
62 }
63 // Increase directory file size by cluster size
64 fileSize_ += 512UL << vol_->clusterSizeShift_;
65 return true;
66}
67//------------------------------------------------------------------------------
68// cache a file's directory entry
69// return pointer to cached entry or null for failure
70dir_t* SdFile::cacheDirEntry(uint8_t action) {
71 if (!SdVolume::cacheRawBlock(dirBlock_, action)) {
72 return NULL;
73 }
74 return SdVolume::cacheBuffer_.dir + dirIndex_;
75}

104uint8_t SdFile::contiguousRange(uint32_t* bgnBlock, uint32_t* endBlock) {
105 // error if no blocks
106 if (firstCluster_ == 0) {
107 return false;
108 }
109
110 for (uint32_t c = firstCluster_; ; c++) {
111 uint32_t next;
112 if (!vol_->fatGet(c, &next)) {
113 return false;
114 }
115
116 // check for contiguous
117 if (next != (c + 1)) {
118 // error if not end of chain
119 if (!vol_->isEOC(next)) {
120 return false;
121 }
122 *bgnBlock = vol_->clusterStartBlock(firstCluster_);
123 *endBlock = vol_->clusterStartBlock(c)
124 + vol_->blocksPerCluster_ - 1;
125 return true;
126 }
127 }
128}

148uint8_t SdFile::createContiguous(SdFile* dirFile,
149 const char* fileName, uint32_t size) {
150 // don't allow zero length file
151 if (size == 0) {
152 return false;
153 }
154 if (!open(dirFile, fileName, O_CREAT | O_EXCL | O_RDWR)) {
155 return false;
156 }
157
158 // calculate number of clusters needed
159 uint32_t count = ((size - 1) >> (vol_->clusterSizeShift_ + 9)) + 1;
160
161 // allocate clusters
162 if (!vol_->allocContiguous(count, &firstCluster_)) {
163 remove();
164 return false;
165 }
166 fileSize_ = size;
167
168 // insure sync() will update dir entry
169 flags_ |= F_FILE_DIR_DIRTY;
170 return sync();
171}

232void SdFile::ls(uint8_t flags, uint8_t indent) {
233 dir_t* p;
234
235 rewind();
236 while ((p = readDirCache())) {
237 // done if past last used entry
238 if (p->name[0] == DIR_NAME_FREE) {
239 break;
240 }
241
242 // skip deleted entry and entries for . and ..
243 if (p->name[0] == DIR_NAME_DELETED || p->name[0] == '.') {
244 continue;
245 }
246
247 // only list subdirectories and files
248 if (!DIR_IS_FILE_OR_SUBDIR(p)) {
249 continue;
250 }
251
252 // print any indent spaces
253 for (int8_t i = 0; i < indent; i++) {
254 Serial.print(' ');
255 }
256
257 // print file name with possible blank fill
258 printDirName(*p, flags & (LS_DATE | LS_SIZE) ? 14 : 0);
259
260 // print modify date/time if requested
261 if (flags & LS_DATE) {
262 printFatDate(p->lastWriteDate);
263 Serial.print(' ');
264 printFatTime(p->lastWriteTime);
265 }
266 // print size if requested
267 if (!DIR_IS_SUBDIR(p) && (flags & LS_SIZE)) {
268 Serial.print(' ');
269 Serial.print(p->fileSize);
270 }
271 Serial.println();
272
273 // list subdirectory content if requested
274 if ((flags & LS_R) && DIR_IS_SUBDIR(p)) {
275 uint16_t index = curPosition() / 32 - 1;
276 SdFile s;
277 if (s.open(this, index, O_READ)) {
278 s.ls(flags, indent + 2);
279 }
280 seekSet(32 * (index + 1));
281 }
282 }
283}

285// format directory name field from a 8.3 name string
286uint8_t SdFile::make83Name(const char* str, uint8_t* name) {
287 uint8_t c;
288 uint8_t n = 7; // max index for part before dot
289 uint8_t i = 0;
290 // blank fill name and extension
291 while (i < 11) {
292 name[i++] = ' ';
293 }
294 i = 0;
295 while ((c = *str++) != '\0') {
296 if (c == '.') {
297 if (n == 10) {
298 return false; // only one dot allowed
299 }
300 n = 10; // max index for full 8.3 name
301 i = 8; // place for extension
302 } else {
303 // illegal FAT characters
304 uint8_t b;
305 #if defined(__AVR__)
306 PGM_P p = PSTR("|<>^+=?/[];,*\"\\");
307 while ((b = pgm_read_byte(p++))) if (b == c) {
308 return false;
309 }
310 #elif defined(__arm__)
311 const uint8_t valid[] = "|<>^+=?/[];,*\"\\";
312 const uint8_t *p = valid;
313 while ((b = *p++)) if (b == c) {
314 return false;
315 }
316 #endif
317 // check size and only allow ASCII printable characters
318 if (i > n || c < 0X21 || c > 0X7E) {
319 return false;
320 }
321 // only upper case allowed in 8.3 names - convert lower to upper
322 name[i++] = c < 'a' || c > 'z' ? c : c + ('A' - 'a');
323 }
324 }
325 // must have a file name, extension is optional
326 return name[0] != ' ';
327}
328//------------------------------------------------------------------------------
341uint8_t SdFile::makeDir(SdFile* dir, const char* dirName) {
342 dir_t d;
343
344 // create a normal file
345 if (!open(dir, dirName, O_CREAT | O_EXCL | O_RDWR)) {
346 return false;
347 }
348
349 // convert SdFile to directory
350 flags_ = O_READ;
351 type_ = FAT_FILE_TYPE_SUBDIR;
352
353 // allocate and zero first cluster
354 if (!addDirCluster()) {
355 return false;
356 }
357
358 // force entry to SD
359 if (!sync()) {
360 return false;
361 }
362
363 // cache entry - should already be in cache due to sync() call
364 dir_t* p = cacheDirEntry(SdVolume::CACHE_FOR_WRITE);
365 if (!p) {
366 return false;
367 }
368
369 // change directory entry attribute
370 p->attributes = DIR_ATT_DIRECTORY;
371
372 // make entry for '.'
373 memcpy(&d, p, sizeof(d));
374 for (uint8_t i = 1; i < 11; i++) {
375 d.name[i] = ' ';
376 }
377 d.name[0] = '.';
378
379 // cache block for '.' and '..'
380 uint32_t block = vol_->clusterStartBlock(firstCluster_);
381 if (!SdVolume::cacheRawBlock(block, SdVolume::CACHE_FOR_WRITE)) {
382 return false;
383 }
384
385 // copy '.' to block
386 memcpy(&SdVolume::cacheBuffer_.dir[0], &d, sizeof(d));
387
388 // make entry for '..'
389 d.name[1] = '.';
390 if (dir->isRoot()) {
391 d.firstClusterLow = 0;
392 d.firstClusterHigh = 0;
393 } else {
394 d.firstClusterLow = dir->firstCluster_ & 0XFFFF;
395 d.firstClusterHigh = dir->firstCluster_ >> 16;
396 }
397 // copy '..' to block
398 memcpy(&SdVolume::cacheBuffer_.dir[1], &d, sizeof(d));
399
400 // set position after '..'
401 curPosition_ = 2 * sizeof(d);
402
403 // write first block
404 return SdVolume::cacheFlush();
405}
406//------------------------------------------------------------------------------
453uint8_t SdFile::open(SdFile* dirFile, const char* fileName, uint8_t oflag) {
454 uint8_t dname[11];
455 dir_t* p;
456
457 // error if already open
458 if (isOpen()) {
459 return false;
460 }
461
462 if (!make83Name(fileName, dname)) {
463 return false;
464 }
465 vol_ = dirFile->vol_;
466 dirFile->rewind();
467
468 // bool for empty entry found
469 uint8_t emptyFound = false;
470
471 // search for file
472 while (dirFile->curPosition_ < dirFile->fileSize_) {
473 uint8_t index = 0XF & (dirFile->curPosition_ >> 5);
474 p = dirFile->readDirCache();
475 if (p == NULL) {
476 return false;
477 }
478
479 if (p->name[0] == DIR_NAME_FREE || p->name[0] == DIR_NAME_DELETED) {
480 // remember first empty slot
481 if (!emptyFound) {
482 emptyFound = true;
483 dirIndex_ = index;
484 dirBlock_ = SdVolume::cacheBlockNumber_;
485 }
486 // done if no entries follow
487 if (p->name[0] == DIR_NAME_FREE) {
488 break;
489 }
490 } else if (!memcmp(dname, p->name, 11)) {
491 // don't open existing file if O_CREAT and O_EXCL
492 if ((oflag & (O_CREAT | O_EXCL)) == (O_CREAT | O_EXCL)) {
493 return false;
494 }
495
496 // open found file
497 return openCachedEntry(0XF & index, oflag);
498 }
499 }
500 // only create file if O_CREAT and O_WRITE
501 if ((oflag & (O_CREAT | O_WRITE)) != (O_CREAT | O_WRITE)) {
502 return false;
503 }
504
505 // cache found slot or add cluster if end of file
506 if (emptyFound) {
507 p = cacheDirEntry(SdVolume::CACHE_FOR_WRITE);
508 if (!p) {
509 return false;
510 }
511 } else {
512 if (dirFile->type_ == FAT_FILE_TYPE_ROOT16) {
513 return false;
514 }
515
516 // add and zero cluster for dirFile - first cluster is in cache for write
517 if (!dirFile->addDirCluster()) {
518 return false;
519 }
520
521 // use first entry in cluster
522 dirIndex_ = 0;
523 p = SdVolume::cacheBuffer_.dir;
524 }
525 // initialize as empty file
526 memset(p, 0, sizeof(dir_t));
527 memcpy(p->name, dname, 11);
528
529 // set timestamps
530 if (dateTime_) {
531 // call user function
532 dateTime_(&p->creationDate, &p->creationTime);
533 } else {
534 // use default date/time
535 p->creationDate = FAT_DEFAULT_DATE;
536 p->creationTime = FAT_DEFAULT_TIME;
537 }
538 p->lastAccessDate = p->creationDate;
539 p->lastWriteDate = p->creationDate;
540 p->lastWriteTime = p->creationTime;
541
542 // force write of entry to SD
543 if (!SdVolume::cacheFlush()) {
544 return false;
545 }
546
547 // open entry in cache
548 return openCachedEntry(dirIndex_, oflag);
549}
550//------------------------------------------------------------------------------
565uint8_t SdFile::open(SdFile* dirFile, uint16_t index, uint8_t oflag) {
566 // error if already open
567 if (isOpen()) {
568 return false;
569 }
570
571 // don't open existing file if O_CREAT and O_EXCL - user call error
572 if ((oflag & (O_CREAT | O_EXCL)) == (O_CREAT | O_EXCL)) {
573 return false;
574 }
575
576 vol_ = dirFile->vol_;
577
578 // seek to location of entry
579 if (!dirFile->seekSet(32 * index)) {
580 return false;
581 }
582
583 // read entry into cache
584 dir_t* p = dirFile->readDirCache();
585 if (p == NULL) {
586 return false;
587 }
588
589 // error if empty slot or '.' or '..'
590 if (p->name[0] == DIR_NAME_FREE ||
591 p->name[0] == DIR_NAME_DELETED || p->name[0] == '.') {
592 return false;
593 }
594 // open cached entry
595 return openCachedEntry(index & 0XF, oflag);
596}
597//------------------------------------------------------------------------------
598// open a cached directory entry. Assumes vol_ is initializes
599uint8_t SdFile::openCachedEntry(uint8_t dirIndex, uint8_t oflag) {
600 // location of entry in cache
601 dir_t* p = SdVolume::cacheBuffer_.dir + dirIndex;
602
603 // write or truncate is an error for a directory or read-only file
604 if (p->attributes & (DIR_ATT_READ_ONLY | DIR_ATT_DIRECTORY)) {
605 if (oflag & (O_WRITE | O_TRUNC)) {
606 return false;
607 }
608 }
609 // remember location of directory entry on SD
610 dirIndex_ = dirIndex;
611 dirBlock_ = SdVolume::cacheBlockNumber_;
612
613 // copy first cluster number for directory fields
614 firstCluster_ = (uint32_t)p->firstClusterHigh << 16;
615 firstCluster_ |= p->firstClusterLow;
616
617 // make sure it is a normal file or subdirectory
618 if (DIR_IS_FILE(p)) {
619 fileSize_ = p->fileSize;
620 type_ = FAT_FILE_TYPE_NORMAL;
621 } else if (DIR_IS_SUBDIR(p)) {
622 if (!vol_->chainSize(firstCluster_, &fileSize_)) {
623 return false;
624 }
625 type_ = FAT_FILE_TYPE_SUBDIR;
626 } else {
627 return false;
628 }
629 // save open flags for read/write
630 flags_ = oflag & (O_ACCMODE | O_SYNC | O_APPEND);
631
632 // set to start of file
633 curCluster_ = 0;
634 curPosition_ = 0;
635
636 // truncate file to zero length if requested
637 if (oflag & O_TRUNC) {
638 return truncate(0);
639 }
640 return true;
641}
642//------------------------------------------------------------------------------
653uint8_t SdFile::openRoot(SdVolume* vol) {
654 // error if file is already open
655 if (isOpen()) {
656 return false;
657 }
658
659 if (vol->fatType() == 16) {
660 type_ = FAT_FILE_TYPE_ROOT16;
661 firstCluster_ = 0;
662 fileSize_ = 32 * vol->rootDirEntryCount();
663 } else if (vol->fatType() == 32) {
664 type_ = FAT_FILE_TYPE_ROOT32;
665 firstCluster_ = vol->rootDirStart();
666 if (!vol->chainSize(firstCluster_, &fileSize_)) {
667 return false;
668 }
669 } else {
670 // volume is not initialized or FAT12
671 return false;
672 }
673 vol_ = vol;
674 // read only
675 flags_ = O_READ;
676
677 // set to start of file
678 curCluster_ = 0;
679 curPosition_ = 0;
680
681 // root has no directory entry
682 dirBlock_ = 0;
683 dirIndex_ = 0;
684 return true;
685}

754//------------------------------------------------------------------------------
769int16_t SdFile::read(void* buf, uint16_t nbyte) {
770 uint8_t* dst = reinterpret_cast<uint8_t*>(buf);
771
772 // error if not open or write only
773 if (!isOpen() || !(flags_ & O_READ)) {
774 return -1;
775 }
776
777 // max bytes left in file
778 if (nbyte > (fileSize_ - curPosition_)) {
779 nbyte = fileSize_ - curPosition_;
780 }
781
782 // amount left to read
783 uint16_t toRead = nbyte;
784 while (toRead > 0) {
785 uint32_t block; // raw device block number
786 uint16_t offset = curPosition_ & 0X1FF; // offset in block
787 if (type_ == FAT_FILE_TYPE_ROOT16) {
788 block = vol_->rootDirStart() + (curPosition_ >> 9);
789 } else {
790 uint8_t blockOfCluster = vol_->blockOfCluster(curPosition_);
791 if (offset == 0 && blockOfCluster == 0) {
792 // start of new cluster
793 if (curPosition_ == 0) {
794 // use first cluster in file
795 curCluster_ = firstCluster_;
796 } else {
797 // get next cluster from FAT
798 if (!vol_->fatGet(curCluster_, &curCluster_)) {
799 return -1;
800 }
801 }
802 }
803 block = vol_->clusterStartBlock(curCluster_) + blockOfCluster;
804 }
805 uint16_t n = toRead;
806
807 // amount to be read from current block
808 if (n > (512 - offset)) {
809 n = 512 - offset;
810 }
811
812 // no buffering needed if n == 512 or user requests no buffering
813 if ((unbufferedRead() || n == 512) &&
814 block != SdVolume::cacheBlockNumber_) {
815 if (!vol_->readData(block, offset, n, dst)) {
816 return -1;
817 }
818 dst += n;
819 } else {
820 // read block to cache and copy data to caller
821 if (!SdVolume::cacheRawBlock(block, SdVolume::CACHE_FOR_READ)) {
822 return -1;
823 }
824 uint8_t* src = SdVolume::cacheBuffer_.data + offset;
825 uint8_t* end = src + n;
826 while (src != end) {
827 *dst++ = *src++;
828 }
829 }
830 curPosition_ += n;
831 toRead -= n;
832 }
833 return nbyte;
834}
835//------------------------------------------------------------------------------
847int8_t SdFile::readDir(dir_t* dir) {
848 int8_t n;
849 // if not a directory file or miss-positioned return an error
850 if (!isDir() || (0X1F & curPosition_)) {
851 return -1;
852 }
853
854 while ((n = read(dir, sizeof(dir_t))) == sizeof(dir_t)) {
855 // last entry if DIR_NAME_FREE
856 if (dir->name[0] == DIR_NAME_FREE) {
857 break;
858 }
859 // skip empty entries and entry for . and ..
860 if (dir->name[0] == DIR_NAME_DELETED || dir->name[0] == '.') {
861 continue;
862 }
863 // return if normal file or subdirectory
864 if (DIR_IS_FILE_OR_SUBDIR(dir)) {
865 return n;
866 }
867 }
868 // error, end of file, or past last entry
869 return n < 0 ? -1 : 0;
870}
871//------------------------------------------------------------------------------
872// Read next directory entry into the cache
873// Assumes file is correctly positioned
874dir_t* SdFile::readDirCache(void) {
875 // error if not directory
876 if (!isDir()) {
877 return NULL;
878 }
879
880 // index of entry in cache
881 uint8_t i = (curPosition_ >> 5) & 0XF;
882
883 // use read to locate and cache block
884 if (read() < 0) {
885 return NULL;
886 }
887
888 // advance to next entry
889 curPosition_ += 31;
890
891 // return pointer to entry
892 return (SdVolume::cacheBuffer_.dir + i);
893}
894//------------------------------------------------------------------------------
909uint8_t SdFile::remove(void) {
910 // free any clusters - will fail if read-only or directory
911 if (!truncate(0)) {
912 return false;
913 }
914
915 // cache directory entry
916 dir_t* d = cacheDirEntry(SdVolume::CACHE_FOR_WRITE);
917 if (!d) {
918 return false;
919 }
920
921 // mark entry deleted
922 d->name[0] = DIR_NAME_DELETED;
923
924 // set this SdFile closed
925 type_ = FAT_FILE_TYPE_CLOSED;
926
927 // write entry to SD
928 return SdVolume::cacheFlush();
929}

972uint8_t SdFile::rmDir(void) {
973 // must be open subdirectory
974 if (!isSubDir()) {
975 return false;
976 }
977
978 rewind();
979
980 // make sure directory is empty
981 while (curPosition_ < fileSize_) {
982 dir_t* p = readDirCache();
983 if (p == NULL) {
984 return false;
985 }
986 // done if past last used entry
987 if (p->name[0] == DIR_NAME_FREE) {
988 break;
989 }
990 // skip empty slot or '.' or '..'
991 if (p->name[0] == DIR_NAME_DELETED || p->name[0] == '.') {
992 continue;
993 }
994 // error not empty
995 if (DIR_IS_FILE_OR_SUBDIR(p)) {
996 return false;
997 }
998 }
999 // convert empty directory to normal file for remove
1000 type_ = FAT_FILE_TYPE_NORMAL;
1001 flags_ |= O_WRITE;
1002 return remove();
1003}
1004//------------------------------------------------------------------------------
1020uint8_t SdFile::rmRfStar(void) {
1021 rewind();
1022 while (curPosition_ < fileSize_) {
1023 SdFile f;
1024
1025 // remember position
1026 uint16_t index = curPosition_ / 32;
1027
1028 dir_t* p = readDirCache();
1029 if (!p) {
1030 return false;
1031 }
1032
1033 // done if past last entry
1034 if (p->name[0] == DIR_NAME_FREE) {
1035 break;
1036 }
1037
1038 // skip empty slot or '.' or '..'
1039 if (p->name[0] == DIR_NAME_DELETED || p->name[0] == '.') {
1040 continue;
1041 }
1042
1043 // skip if part of long file name or volume label in root
1044 if (!DIR_IS_FILE_OR_SUBDIR(p)) {
1045 continue;
1046 }
1047
1048 if (!f.open(this, index, O_READ)) {
1049 return false;
1050 }
1051 if (f.isSubDir()) {
1052 // recursively delete
1053 if (!f.rmRfStar()) {
1054 return false;
1055 }
1056 } else {
1057 // ignore read-only
1058 f.flags_ |= O_WRITE;
1059 if (!f.remove()) {
1060 return false;
1061 }
1062 }
1063 // position to next entry if required
1064 if (curPosition_ != (32u * (index + 1))) {
1065 if (!seekSet(32u * (index + 1))) {
1066 return false;
1067 }
1068 }
1069 }
1070 // don't try to delete root
1071 if (isRoot()) {
1072 return true;
1073 }
1074 return rmDir();

1085uint8_t SdFile::seekSet(uint32_t pos) {
1086 // error if file not open or seek past end of file
1087 if (!isOpen() || pos > fileSize_) {
1088 return false;
1089 }
1090
1091 if (type_ == FAT_FILE_TYPE_ROOT16) {
1092 curPosition_ = pos;
1093 return true;
1094 }
1095 if (pos == 0) {
1096 // set position to start of file
1097 curCluster_ = 0;
1098 curPosition_ = 0;
1099 return true;
1100 }
1101 // calculate cluster index for cur and new position
1102 uint32_t nCur = (curPosition_ - 1) >> (vol_->clusterSizeShift_ + 9);
1103 uint32_t nNew = (pos - 1) >> (vol_->clusterSizeShift_ + 9);
1104
1105 if (nNew < nCur || curPosition_ == 0) {
1106 // must follow chain from first cluster
1107 curCluster_ = firstCluster_;
1108 } else {
1109 // advance from curPosition
1110 nNew -= nCur;
1111 }
1112 while (nNew--) {
1113 if (!vol_->fatGet(curCluster_, &curCluster_)) {
1114 return false;
1115 }
1116 }
1117 curPosition_ = pos;
1118 return true;
1119}

1133 // only allow open files and directories
1134 if (!isOpen()) {
1135 return false;
1136 }
1137
1138 if (flags_ & F_FILE_DIR_DIRTY) {
1139 dir_t* d = cacheDirEntry(SdVolume::CACHE_FOR_WRITE);
1140 if (!d) {
1141 return false;
1142 }
1143
1144 // do not set filesize for dir files
1145 if (!isDir()) {
1146 d->fileSize = fileSize_;
1147 }
1148
1149 // update first cluster fields
1150 d->firstClusterLow = firstCluster_ & 0XFFFF;
1151 d->firstClusterHigh = firstCluster_ >> 16;
1152
1153 // set modify time if user supplied a callback date/time function
1154 if (dateTime_) {
1155 dateTime_(&d->lastWriteDate, &d->lastWriteTime);
1156 d->lastAccessDate = d->lastWriteDate;

1159 flags_ &= ~F_FILE_DIR_DIRTY;
1160 }

1164 }

1253uint8_t SdFile::truncate(uint32_t length) {
1254 // error if not a normal file or read-only
1255 if (!isFile() || !(flags_ & O_WRITE)) {
1256 return false;
1257 }
1258
1259 // error if length is greater than current size
1260 if (length > fileSize_) {
1261 return false;
1262 }
1263
1264 // fileSize and length are zero - nothing to do
1265 if (fileSize_ == 0) {
1266 return true;
1267 }
1268
1269 // remember position for seek after truncation
1270 uint32_t newPos = curPosition_ > length ? length : curPosition_;
1271
1272 // position to last cluster in truncated file
1273 if (!seekSet(length)) {
1274 return false;
1275 }
1276
1277 if (length == 0) {
1278 // free all clusters
1279 if (!vol_->freeChain(firstCluster_)) {
1280 return false;
1281 }
1282 firstCluster_ = 0;
1283 } else {
1284 uint32_t toFree;
1285 if (!vol_->fatGet(curCluster_, &toFree)) {
1286 return false;
1287 }
1288
1289 if (!vol_->isEOC(toFree)) {
1290 // free extra clusters
1291 if (!vol_->freeChain(toFree)) {
1292 return false;
1293 }
1294
1295 // current cluster is end of chain
1296 if (!vol_->fatPutEOC(curCluster_)) {
1297 return false;
1298 }
1299 }
1300 }
1301 fileSize_ = length;
1302
1303 // need to update directory entry
1304 flags_ |= F_FILE_DIR_DIRTY;
1305
1306 if (!sync()) {
1307 return false;
1308 }
1309
1310 // set file to correct position
1311 return seekSet(newPos);
1312}

1330size_t SdFile::write(const void* buf, uint16_t nbyte) {
1331 // convert void* to uint8_t* - must be before goto statements
1332 const uint8_t* src = reinterpret_cast<const uint8_t*>(buf);
1333
1334 // number of bytes left to write - must be before goto statements
1335 uint16_t nToWrite = nbyte;
1336 // if blocking writes should be used
1337 uint8_t blocking = (flags_ & F_FILE_NON_BLOCKING_WRITE) == 0x00;
1338
1339 // error if not a normal file or is read-only
1340 if (!isFile() || !(flags_ & O_WRITE)) {
1341 goto writeErrorReturn;
1342 }
1343
1344 // seek to end of file if append flag
1345 if ((flags_ & O_APPEND) && curPosition_ != fileSize_) {
1346 if (!seekEnd()) {
1347 goto writeErrorReturn;
1348 }
1349 }
1350
1351 while (nToWrite > 0) {
1352 uint8_t blockOfCluster = vol_->blockOfCluster(curPosition_);
1353 uint16_t blockOffset = curPosition_ & 0X1FF;
1354 if (blockOfCluster == 0 && blockOffset == 0) {
1355 // start of new cluster
1356 if (curCluster_ == 0) {
1357 if (firstCluster_ == 0) {
1358 // allocate first cluster of file
1359 if (!addCluster()) {
1360 goto writeErrorReturn;
1361 }
1362 } else {
1363 curCluster_ = firstCluster_;
1364 }
1365 } else {
1366 uint32_t next;
1367 if (!vol_->fatGet(curCluster_, &next)) {
1368 return false;
1369 }
1370 if (vol_->isEOC(next)) {
1371 // add cluster if at end of chain
1372 if (!addCluster()) {
1373 goto writeErrorReturn;
1374 }
1375 } else {
1376 curCluster_ = next;
1377 }
1378 }
1379 }
1380 // max space in block
1381 uint16_t n = 512 - blockOffset;
1382
1383 // lesser of space and amount to write
1384 if (n > nToWrite) {
1385 n = nToWrite;
1386 }
1387
1388 // block for data write
1389 uint32_t block = vol_->clusterStartBlock(curCluster_) + blockOfCluster;
1390 if (n == 512) {
1391 // full block - don't need to use cache
1392 // invalidate cache if block is in cache
1393 if (SdVolume::cacheBlockNumber_ == block) {
1394 SdVolume::cacheBlockNumber_ = 0XFFFFFFFF;
1395 }
1396 if (!vol_->writeBlock(block, src, blocking)) {
1397 goto writeErrorReturn;
1398 }
1399 src += 512;
1400 } else {
1401 if (blockOffset == 0 && curPosition_ >= fileSize_) {
1402 // start of new block don't need to read into cache
1403 if (!SdVolume::cacheFlush()) {
1404 goto writeErrorReturn;
1405 }
1406 SdVolume::cacheBlockNumber_ = block;
1407 SdVolume::cacheSetDirty();
1408 } else {
1409 // rewrite part of block
1410 if (!SdVolume::cacheRawBlock(block, SdVolume::CACHE_FOR_WRITE)) {
1411 goto writeErrorReturn;
1412 }
1413 }
1414 uint8_t* dst = SdVolume::cacheBuffer_.data + blockOffset;
1415 uint8_t* end = dst + n;
1416 while (dst != end) {
1417 *dst++ = *src++;
1418 }
1419 }
1420 nToWrite -= n;
1421 curPosition_ += n;
1422 }
1423 if (curPosition_ > fileSize_) {
1424 // update fileSize and insure sync will update dir entry
1425 fileSize_ = curPosition_;
1426 flags_ |= F_FILE_DIR_DIRTY;
1427 } else if (dateTime_ && nbyte) {
1428 // insure sync will update modified date and time
1429 flags_ |= F_FILE_DIR_DIRTY;
1430 }
1431
1432 if (flags_ & O_SYNC) {
1433 if (!sync()) {
1434 goto writeErrorReturn;
1435 }
1436 }
1437 return nbyte;
1438
1439writeErrorReturn:
1440 // return for write error
1441 //writeError = true;
1442 setWriteError();
1443 return 0;
1444}

1492int SdFile::availableForWrite() {
1493 if (!isFile() || !(flags_ & O_WRITE)) {
1494 return 0;
1495 }
1496
1497 // seek to end of file if append flag
1498 if ((flags_ & O_APPEND) && curPosition_ != fileSize_) {
1499 if (!seekEnd()) {
1500 return 0;
1501 }
1502 }
1503
1504 if (vol_->isBusy()) {
1505 return 0;
1506 }
1507
1508 if (flags_ & F_FILE_CLUSTER_ADDED) {
1509 // new cluster added, trigger a non-blocking sync
1510 sync(0);
1511 flags_ &= ~F_FILE_CLUSTER_ADDED;
1512 return 0;
1513 }
1514
1515 if (vol_->isCacheMirrorBlockDirty()) {
1516 // cache mirror block is dirty, trigger a non-blocking sync
1517 vol_->cacheMirrorBlockFlush(0);
1518 return 0;
1519 }
1520
1521 flags_ |= F_FILE_NON_BLOCKING_WRITE;
1522
1523 uint16_t blockOffset = curPosition_ & 0X1FF;
1524 uint16_t n = 512 - blockOffset;
1525
1526 return n;
