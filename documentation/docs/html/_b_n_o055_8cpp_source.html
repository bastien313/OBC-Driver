81bool BNO055::check(){
82//Check we have communication link with the chip
83 setpage(0);
84 uint8_t rx = readRegister(BNO055_CHIP_ID_ADDR);
85 uint8_t rawdata[7];
86 if (rx != 0xA0) return false;
87//Grab the chip ID and software versions
88 readRegister(BNO055_CHIP_ID_ADDR, rawdata, 7);
89 ID.id = rawdata[0];
90 ID.accel = rawdata[1];
91 ID.mag = rawdata[2];
92 ID.gyro = rawdata[3];
93 ID.sw[0] = rawdata[4];
94 ID.sw[1] = rawdata[5];
95 ID.bootload = rawdata[6];
96 setpage(1);
97 readRegister(BNO055_UNIQUE_ID_ADDR, (uint8_t*)ID.serial, 16);
98 setpage(0);
99 return true;
