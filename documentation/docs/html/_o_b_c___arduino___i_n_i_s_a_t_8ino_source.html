14void loop() {
15 char strFloat[20];
16 char command = 0;
17
18 if(OBC.readCom(&command,1)){
19 switch(command){
20
21 case 'A': // A check
22 OBC.writeCom((char*)"AOBC check : OK@");
23 break;
24
25 case 'B': // B (PARMETRES BATTERIES)
26 EPS_dataOut();
27 break;
28
29 case 'C': // C (TEMPERATURE)
30 OBC.BNO.get_temp();
31 OBC.writeCom((char*)"C");
32 dtostrf(OBC.BNO.temperature, 7, 3, strFloat);
33 OBC.writeCom(strFloat);
34 OBC.writeCom((char*)"@");
35 break;
36
37 case 'D': // D (ALTITUDE)
38 MLP_altitude();
39 break;
40
41 case 'E': // E (PRESSION)
42 MLP_pression();
43 break;
44
45 case 'F': // F (ANGLE EULER)
46 BNO_euler();
47 break;
48
49 case 'G': // G (QUATERNION)
50 BNO_quaternion();
51 break;
52
53 case 'H': // H (VITESSE ANGULAIRE)
54 BNO_angularSpeed();
55 break;
56
57 case 'I': // I (ACCELERATION ANGULAIRE)
58 BNO_accel();
59 break;
60
61 case 'J': // J (CHAMP MAGNETIQUE)
62 BNO_mag();
63 break;
64
65 case 'K': // K (ACCELERATION LINEAIRE)
66 BNO_lia();
67 break;
68
69 case 'L': // L (GRAVITE)
70 BNO_gravity();
71 break;
72
73 case 'M': // M (LUMINOSITE)
74 solar();
75 break;
76
77 case 'N': // O (GNSS EXTRACT)
78 NEMA_dataWrite();
79 break;
80
81 case 'O': // O (GNSS RAW)
82 NMEA_bufferWrite();
83 break;
84
85 case 'R': // R (TEMPERATURES EXTERIEURES - ADC)
86
87 break;
88
89
90 case 'V': // V (GESTION INTERNE)
91 OBC.writeCom((char*)"V2@");
92 break;
93
94 case 'Z': // Z FIN DE COMMUNICATION
95 OBC.writeCom((char*)"Z");
96 break;
97 }
98 }
99}

364void NEMA_dataWrite(void){
365 char bufferOut[256];
366 OBC.GNSS.fill_NMEA_buffer();
367
368 if(strlen((char*)OBC.GNSS.NMEA_Buffer) == 0){
369 OBC.writeCom((char*)"PNo data@");
370 return;
371 }
372
373 //Serial.print((char*)OBC.GNSS.NMEA_Buffer);
374
375 int hTime, mTime, sTime;
376 float latitude, longitude;
377 GxGGA_parser GGA;
378 if(OBC.GNSS.getGGAdata(&GGA, false)){
379 latitude = INISAT_GNSS::DecimalDegreesToGpsCoordinate(GGA.latitude, GGA.ns);
380 longitude = INISAT_GNSS::DecimalDegreesToGpsCoordinate(GGA.longitude, GGA.ew);
381 hTime = INISAT_GNSS::calcHour(GGA.fix);
382 mTime = INISAT_GNSS::calcMinute(GGA.fix);
383 sTime = INISAT_GNSS::calcSecond(GGA.fix);
384
385 //GGA.debugPrint();
386
387 if(GGA.filedIsPresent(GxGGA_parser::LATITUDE_POS) && GGA.filedIsPresent(GxGGA_parser::LONGITUDE_POS)){ // Data is valid
388 sendNMEAData(hTime, mTime, sTime, latitude, GGA.ns, longitude, GGA.ew, GGA.satNumber);
389 return; // data valid we skip
390 }
391 }
392
393 GxGLL_parser GLL;
394 if(OBC.GNSS.getGLLdata(&GLL, false)){
395 latitude = INISAT_GNSS::DecimalDegreesToGpsCoordinate(GLL.latitude, GLL.ns);
396 longitude = INISAT_GNSS::DecimalDegreesToGpsCoordinate(GLL.longitude, GLL.ew);
397 hTime = INISAT_GNSS::calcHour(GLL.fix);
398 mTime = INISAT_GNSS::calcMinute(GLL.fix);
399 sTime = INISAT_GNSS::calcSecond(GLL.fix);
400
401 //GLL.debugPrint();
402
403 if(GLL.filedIsPresent(GxGLL_parser::LATITUDE_POS) && GLL.filedIsPresent(GxGLL_parser::LONGITUDE_POS)){ // Data is valid
404 sendNMEAData(hTime, mTime, sTime, latitude, GLL.ns, longitude, GLL.ew, 0);
405 return; // data valid we skip
406 }
407 }
408 // No good data available
409 OBC.writeCom((char*)"PNo data@");
410}
