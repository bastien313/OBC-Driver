111static inline uint16_t FAT_DATE(uint16_t year, uint8_t month, uint8_t day) {
112 return (year - 1980) << 9 | month << 5 | day;
113}
115static inline uint16_t FAT_YEAR(uint16_t fatDate) {
116 return 1980 + (fatDate >> 9);
117}
119static inline uint8_t FAT_MONTH(uint16_t fatDate) {
120 return (fatDate >> 5) & 0XF;
121}
123static inline uint8_t FAT_DAY(uint16_t fatDate) {
124 return fatDate & 0X1F;
125}
127static inline uint16_t FAT_TIME(uint8_t hour, uint8_t minute, uint8_t second) {
128 return hour << 11 | minute << 5 | second >> 1;
129}
131static inline uint8_t FAT_HOUR(uint16_t fatTime) {
132 return fatTime >> 11;
133}
135static inline uint8_t FAT_MINUTE(uint16_t fatTime) {
136 return (fatTime >> 5) & 0X3F;
137}
139static inline uint8_t FAT_SECOND(uint16_t fatTime) {
140 return 2 * (fatTime & 0X1F);
141}
