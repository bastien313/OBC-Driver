136int16_t ADS1015::readADC_SingleEnded(uint8_t channel)
137{
138 if (channel > 3) {
139 return 0;
140 }
141
142 // Start with default values
143 uint16_t config = ADS1015_REG_CONFIG_CQUE_NONE | // Disable the comparator (default val)
144 ADS1015_REG_CONFIG_CLAT_NONLAT | // Non-latching (default val)
145 ADS1015_REG_CONFIG_CPOL_ACTVLOW | // Alert/Rdy active low (default val)
146 ADS1015_REG_CONFIG_CMODE_TRAD | // Traditional comparator (default val)
147 ADS1015_REG_CONFIG_DR_3300SPS | // 860(ADS1115) samples per second
148 ADS1015_REG_CONFIG_MODE_SINGLE; // Single-shot mode (default)
149
150 // Set PGA/voltage range
151 config |= m_gain;
152
153 // Set single-ended input channel
154 switch (channel) {
155 case (0):
156 config |= ADS1015_REG_CONFIG_MUX_SINGLE_0;
157 break;
158 case (1):
159 config |= ADS1015_REG_CONFIG_MUX_SINGLE_1;
160 break;
161 case (2):
162 config |= ADS1015_REG_CONFIG_MUX_SINGLE_2;
163 break;
164 case (3):
165 config |= ADS1015_REG_CONFIG_MUX_SINGLE_3;
166 break;
167 }
168
169 // Set 'start single-conversion' bit
170 config |= ADS1015_REG_CONFIG_OS_SINGLE;
171
172 // Write config register to the ADC
173 writeRegister(m_i2cAddress, ADS1015_REG_POINTER_CONFIG, config);
174
175 // Wait for the conversion to complete
176 waitEndOfConversion();
177
178 // Read the conversion results
179 uint16_t rawResult = readRegister(m_i2cAddress, ADS1015_REG_POINTER_CONVERT);
180 return *((int16_t*)(&rawResult));
181// return 1;
182}
