191void MPL3115A2::setOversampleRate(char sampleRate)
192{
193 if(sampleRate > 7)
194 sampleRate = 7; // OS cannot be larger than 0b.0111

197
198 char temp = i2cRead(CTRL_REG1); // Read current settings
199 temp &= 0xC7; // Clear out old OS bits
200 temp |= sampleRate; // Mask in new OS bits
