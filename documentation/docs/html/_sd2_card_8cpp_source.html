36static void spiSend(uint8_t b) {
37 #ifndef USE_SPI_LIB
38 SPDR = b;
39 while (!(SPSR & (1 << SPIF)))
40 ;
41 #else
42 SDCARD_SPI.transfer(b);
43 #endif
44}
46static uint8_t spiRec(void) {
47 #ifndef USE_SPI_LIB
48 spiSend(0XFF);
49 return SPDR;
50 #else
51 return SDCARD_SPI.transfer(0xFF);
52 #endif
53}
54#else // SOFTWARE_SPI

57#define nop asm volatile ("nop\n\t")
58//------------------------------------------------------------------------------
60uint8_t spiRec(void) {
61 uint8_t data = 0;
62 // no interrupts during byte receive - about 8 us
63 cli();
64 // output pin high - like sending 0XFF
65 fastDigitalWrite(SPI_MOSI_PIN, HIGH);
66
67 for (uint8_t i = 0; i < 8; i++) {
68 fastDigitalWrite(SPI_SCK_PIN, HIGH);
69
70 // adjust so SCK is nice
71 nop;
72 nop;
73
74 data <<= 1;
75
76 if (fastDigitalRead(SPI_MISO_PIN)) {
77 data |= 1;
78 }
79
80 fastDigitalWrite(SPI_SCK_PIN, LOW);
81 }
82 // enable interrupts
83 sei();
84 return data;
85}
86//------------------------------------------------------------------------------
88void spiSend(uint8_t data) {
89 // no interrupts during byte send - about 8 us
90 cli();
91 for (uint8_t i = 0; i < 8; i++) {
92 fastDigitalWrite(SPI_SCK_PIN, LOW);
93
94 fastDigitalWrite(SPI_MOSI_PIN, data & 0X80);
95
96 data <<= 1;
97
98 fastDigitalWrite(SPI_SCK_PIN, HIGH);
99 }
100 // hold SCK high for a few ns
101 nop;
102 nop;
103 nop;
104 nop;
105
106 fastDigitalWrite(SPI_SCK_PIN, LOW);
107 // enable interrupts
108 sei();
109}

112// send command and return error code. Return zero for OK
113uint8_t Sd2Card::cardCommand(uint8_t cmd, uint32_t arg) {
114 // end read if in partialBlockRead mode
115 readEnd();
116
117 // select card
118 chipSelectLow();
119
120 // wait up to 300 ms if busy
121 waitNotBusy(300);
122
123 // send command
124 spiSend(cmd | 0x40);
125
126 // send argument
127 for (int8_t s = 24; s >= 0; s -= 8) {
128 spiSend(arg >> s);
129 }
130
131 // send CRC
132 uint8_t crc = 0XFF;
133 if (cmd == CMD0) {
134 crc = 0X95; // correct crc for CMD0 with arg 0
135 }
136 if (cmd == CMD8) {
137 crc = 0X87; // correct crc for CMD8 with arg 0X1AA
138 }
139 spiSend(crc);
140
141 // wait for response
142 for (uint8_t i = 0; ((status_ = spiRec()) & 0X80) && i != 0XFF; i++)
143 ;
144 return status_;
145}

153uint32_t Sd2Card::cardSize(void) {
154 csd_t csd;
155 if (!readCSD(&csd)) {
156 return 0;
157 }
158 if (csd.v1.csd_ver == 0) {
159 uint8_t read_bl_len = csd.v1.read_bl_len;
160 uint16_t c_size = (csd.v1.c_size_high << 10)
161 | (csd.v1.c_size_mid << 2) | csd.v1.c_size_low;
162 uint8_t c_size_mult = (csd.v1.c_size_mult_high << 1)
163 | csd.v1.c_size_mult_low;
164 return (uint32_t)(c_size + 1) << (c_size_mult + read_bl_len - 7);
165 } else if (csd.v2.csd_ver == 1) {
166 uint32_t c_size = ((uint32_t)csd.v2.c_size_high << 16)
167 | (csd.v2.c_size_mid << 8) | csd.v2.c_size_low;
168 return (c_size + 1) << 10;
169 } else {
170 error(SD_CARD_ERROR_BAD_CSD);
171 return 0;
172 }
173}

210uint8_t Sd2Card::erase(uint32_t firstBlock, uint32_t lastBlock) {
211 if (!eraseSingleBlockEnable()) {
212 error(SD_CARD_ERROR_ERASE_SINGLE_BLOCK);
213 goto fail;
214 }
215 if (type_ != SD_CARD_TYPE_SDHC) {
216 firstBlock <<= 9;
217 lastBlock <<= 9;
218 }
219 if (cardCommand(CMD32, firstBlock)
220 || cardCommand(CMD33, lastBlock)
221 || cardCommand(CMD38, 0)) {
222 error(SD_CARD_ERROR_ERASE);
223 goto fail;
224 }
225 if (!waitNotBusy(SD_ERASE_TIMEOUT)) {
226 error(SD_CARD_ERROR_ERASE_TIMEOUT);
227 goto fail;
228 }
229 chipSelectHigh();
230 return true;
231
232fail:
233 chipSelectHigh();
234 return false;
235}

257uint8_t Sd2Card::init(uint8_t sckRateID, uint8_t chipSelectPin) {
258 errorCode_ = inBlock_ = partialBlockRead_ = type_ = 0;
259 chipSelectPin_ = chipSelectPin;
260 // 16-bit init start time allows over a minute
261 unsigned int t0 = millis();
262 uint32_t arg;
263
264 // set pin modes
265 pinMode(chipSelectPin_, OUTPUT);
266 digitalWrite(chipSelectPin_, HIGH);
267 #ifndef USE_SPI_LIB
268 pinMode(SPI_MISO_PIN, INPUT);
269 pinMode(SPI_MOSI_PIN, OUTPUT);
270 pinMode(SPI_SCK_PIN, OUTPUT);
271 #endif
272
273 #ifndef SOFTWARE_SPI
274 #ifndef USE_SPI_LIB
275 // SS must be in output mode even it is not chip select
276 pinMode(SS_PIN, OUTPUT);
277 digitalWrite(SS_PIN, HIGH); // disable any SPI device using hardware SS pin
278 // Enable SPI, Master, clock rate f_osc/128
279 SPCR = (1 << SPE) | (1 << MSTR) | (1 << SPR1) | (1 << SPR0);
280 // clear double speed
281 SPSR &= ~(1 << SPI2X);
282 #else // USE_SPI_LIB
283 SDCARD_SPI.begin();
284 settings = SPISettings(250000, MSBFIRST, SPI_MODE0);
285 #endif // USE_SPI_LIB
286 #endif // SOFTWARE_SPI
287
288 // must supply min of 74 clock cycles with CS high.
289 #ifdef USE_SPI_LIB
290 SDCARD_SPI.beginTransaction(settings);
291 #endif
292 for (uint8_t i = 0; i < 10; i++) {
293 spiSend(0XFF);
294 }
295 #ifdef USE_SPI_LIB
296 SDCARD_SPI.endTransaction();
297 #endif
298
299 chipSelectLow();
300
301 // command to go idle in SPI mode
302 while ((status_ = cardCommand(CMD0, 0)) != R1_IDLE_STATE) {
303 unsigned int d = millis() - t0;
304 if (d > SD_INIT_TIMEOUT) {
305 error(SD_CARD_ERROR_CMD0);
306 goto fail;
307 }
308 }
309 // check SD version
310 if ((cardCommand(CMD8, 0x1AA) & R1_ILLEGAL_COMMAND)) {
311 type(SD_CARD_TYPE_SD1);
312 } else {
313 // only need last byte of r7 response
314 for (uint8_t i = 0; i < 4; i++) {
315 status_ = spiRec();
316 }
317 if (status_ != 0XAA) {
318 error(SD_CARD_ERROR_CMD8);
319 goto fail;
320 }
321 type(SD_CARD_TYPE_SD2);
322 }
323 // initialize card and send host supports SDHC if SD2
324 arg = type() == SD_CARD_TYPE_SD2 ? 0X40000000 : 0;
325
326 while ((status_ = cardAcmd(ACMD41, arg)) != R1_READY_STATE) {
327 // check for timeout
328 unsigned int d = millis() - t0;
329 if (d > SD_INIT_TIMEOUT) {
330 error(SD_CARD_ERROR_ACMD41);
331 goto fail;
332 }
333 }
334 // if SD2 read OCR register to check for SDHC card
335 if (type() == SD_CARD_TYPE_SD2) {
336 if (cardCommand(CMD58, 0)) {
337 error(SD_CARD_ERROR_CMD58);
338 goto fail;
339 }
340 if ((spiRec() & 0XC0) == 0XC0) {
341 type(SD_CARD_TYPE_SDHC);
342 }
343 // discard rest of ocr - contains allowed voltage range
344 for (uint8_t i = 0; i < 3; i++) {
345 spiRec();
346 }
347 }
348 chipSelectHigh();
349
350 #ifndef SOFTWARE_SPI
351 return setSckRate(sckRateID);
352 #else // SOFTWARE_SPI
353 return true;
354 #endif // SOFTWARE_SPI
355
356fail:
357 chipSelectHigh();
358 return false;
359}

402uint8_t Sd2Card::readData(uint32_t block,
403 uint16_t offset, uint16_t count, uint8_t* dst) {
404 if (count == 0) {
405 return true;
406 }
407 if ((count + offset) > 512) {
408 goto fail;
409 }
410 if (!inBlock_ || block != block_ || offset < offset_) {
411 block_ = block;
412 // use address if not SDHC card
413 if (type() != SD_CARD_TYPE_SDHC) {
414 block <<= 9;
415 }
416 if (cardCommand(CMD17, block)) {
417 error(SD_CARD_ERROR_CMD17);
418 goto fail;
419 }
420 if (!waitStartBlock()) {
421 goto fail;
422 }
423 offset_ = 0;
424 inBlock_ = 1;
425 }
426
427 #ifdef OPTIMIZE_HARDWARE_SPI
428 // start first spi transfer
429 SPDR = 0XFF;
430
431 // skip data before offset
432 for (; offset_ < offset; offset_++) {
433 while (!(SPSR & (1 << SPIF)))
434 ;
435 SPDR = 0XFF;
436 }
437 // transfer data
438 n = count - 1;
439 for (uint16_t i = 0; i < n; i++) {
440 while (!(SPSR & (1 << SPIF)))
441 ;
442 dst[i] = SPDR;
443 SPDR = 0XFF;
444 }
445 // wait for last byte
446 while (!(SPSR & (1 << SPIF)))
447 ;
448 dst[n] = SPDR;
449
450 #else // OPTIMIZE_HARDWARE_SPI
451
452 // skip data before offset
453 for (; offset_ < offset; offset_++) {
454 spiRec();
455 }
456 // transfer data
457 for (uint16_t i = 0; i < count; i++) {
458 dst[i] = spiRec();
459 }
460 #endif // OPTIMIZE_HARDWARE_SPI
461
462 offset_ += count;
463 if (!partialBlockRead_ || offset_ >= 512) {
464 // read rest of data, checksum and set chip select high
465 readEnd();
466 }
467 return true;
468
469fail:
470 chipSelectHigh();
471 return false;
472}
473//------------------------------------------------------------------------------
475void Sd2Card::readEnd(void) {
476 if (inBlock_) {
477 // skip data and crc
478 #ifdef OPTIMIZE_HARDWARE_SPI
479 // optimize skip for hardware
480 SPDR = 0XFF;
481 while (offset_++ < 513) {
482 while (!(SPSR & (1 << SPIF)))
483 ;
484 SPDR = 0XFF;
485 }
486 // wait for last crc byte
487 while (!(SPSR & (1 << SPIF)))
488 ;
489 #else // OPTIMIZE_HARDWARE_SPI
490 while (offset_++ < 514) {
491 spiRec();
492 }
493 #endif // OPTIMIZE_HARDWARE_SPI
494 chipSelectHigh();
495 inBlock_ = 0;
496 }
497}
498//------------------------------------------------------------------------------
500uint8_t Sd2Card::readRegister(uint8_t cmd, void* buf) {
501 uint8_t* dst = reinterpret_cast<uint8_t*>(buf);
502 if (cardCommand(cmd, 0)) {
503 error(SD_CARD_ERROR_READ_REG);
504 goto fail;
505 }
506 if (!waitStartBlock()) {
507 goto fail;
508 }
509 // transfer data
510 for (uint16_t i = 0; i < 16; i++) {
511 dst[i] = spiRec();
512 }
513 spiRec(); // get first crc byte
514 spiRec(); // get second crc byte
515 chipSelectHigh();
516 return true;
517
518fail:
519 chipSelectHigh();
520 return false;
521}
522//------------------------------------------------------------------------------
535uint8_t Sd2Card::setSckRate(uint8_t sckRateID) {
536 if (sckRateID > 6) {
537 error(SD_CARD_ERROR_SCK_RATE);
538 return false;
539 }
540 #ifndef USE_SPI_LIB
541 // see avr processor datasheet for SPI register bit definitions
542 if ((sckRateID & 1) || sckRateID == 6) {
543 SPSR &= ~(1 << SPI2X);
544 } else {
545 SPSR |= (1 << SPI2X);
546 }
547 SPCR &= ~((1 << SPR1) | (1 << SPR0));
548 SPCR |= (sckRateID & 4 ? (1 << SPR1) : 0)
549 | (sckRateID & 2 ? (1 << SPR0) : 0);
550 #else // USE_SPI_LIB
551 switch (sckRateID) {
552 case 0: settings = SPISettings(25000000, MSBFIRST, SPI_MODE0); break;
553 case 1: settings = SPISettings(4000000, MSBFIRST, SPI_MODE0); break;
554 case 2: settings = SPISettings(2000000, MSBFIRST, SPI_MODE0); break;
555 case 3: settings = SPISettings(1000000, MSBFIRST, SPI_MODE0); break;
556 case 4: settings = SPISettings(500000, MSBFIRST, SPI_MODE0); break;
557 case 5: settings = SPISettings(250000, MSBFIRST, SPI_MODE0); break;
558 default: settings = SPISettings(125000, MSBFIRST, SPI_MODE0);
559 }
560 #endif // USE_SPI_LIB
561 return true;
562}

572// wait for card to go not busy
573uint8_t Sd2Card::waitNotBusy(unsigned int timeoutMillis) {
574 unsigned int t0 = millis();
575 unsigned int d;
576 do {
577 if (spiRec() == 0XFF) {
578 return true;
579 }
580 d = millis() - t0;
581 } while (d < timeoutMillis);
582 return false;
583}
584//------------------------------------------------------------------------------
586uint8_t Sd2Card::waitStartBlock(void) {
587 unsigned int t0 = millis();
588 while ((status_ = spiRec()) == 0XFF) {
589 unsigned int d = millis() - t0;
590 if (d > SD_READ_TIMEOUT) {
591 error(SD_CARD_ERROR_READ_TIMEOUT);
592 goto fail;
593 }
594 }
595 if (status_ != DATA_START_BLOCK) {
596 error(SD_CARD_ERROR_READ);
597 goto fail;
598 }
599 return true;
600
601fail:
602 chipSelectHigh();
603 return false;
604}

615uint8_t Sd2Card::writeBlock(uint32_t blockNumber, const uint8_t* src, uint8_t blocking) {
616 #if SD_PROTECT_BLOCK_ZERO
617 // don't allow write to first block
618 if (blockNumber == 0) {
619 error(SD_CARD_ERROR_WRITE_BLOCK_ZERO);
620 goto fail;
621 }
622 #endif // SD_PROTECT_BLOCK_ZERO
623
624 // use address if not SDHC card
625 if (type() != SD_CARD_TYPE_SDHC) {
626 blockNumber <<= 9;
627 }
628 if (cardCommand(CMD24, blockNumber)) {
629 error(SD_CARD_ERROR_CMD24);
630 goto fail;
631 }
632 if (!writeData(DATA_START_BLOCK, src)) {
633 goto fail;
634 }
635 if (blocking) {
636 // wait for flash programming to complete
637 if (!waitNotBusy(SD_WRITE_TIMEOUT)) {
638 error(SD_CARD_ERROR_WRITE_TIMEOUT);
639 goto fail;
640 }
641 // response is r2 so get and check two bytes for nonzero
642 if (cardCommand(CMD13, 0) || spiRec()) {
643 error(SD_CARD_ERROR_WRITE_PROGRAMMING);
644 goto fail;
645 }
646 }
647 chipSelectHigh();
648 return true;
649
650fail:
651 chipSelectHigh();
652 return false;
653}
654//------------------------------------------------------------------------------
656uint8_t Sd2Card::writeData(const uint8_t* src) {
657 // wait for previous write to finish
658 if (!waitNotBusy(SD_WRITE_TIMEOUT)) {
659 error(SD_CARD_ERROR_WRITE_MULTIPLE);
660 chipSelectHigh();
661 return false;
662 }
663 return writeData(WRITE_MULTIPLE_TOKEN, src);
664}
665//------------------------------------------------------------------------------
666// send one block of data for write block or write multiple blocks
667uint8_t Sd2Card::writeData(uint8_t token, const uint8_t* src) {
668 #ifdef OPTIMIZE_HARDWARE_SPI
669
670 // send data - optimized loop
671 SPDR = token;
672
673 // send two byte per iteration
674 for (uint16_t i = 0; i < 512; i += 2) {
675 while (!(SPSR & (1 << SPIF)))
676 ;
677 SPDR = src[i];
678 while (!(SPSR & (1 << SPIF)))
679 ;
680 SPDR = src[i + 1];
681 }
682
683 // wait for last data byte
684 while (!(SPSR & (1 << SPIF)))
685 ;
686
687 #else // OPTIMIZE_HARDWARE_SPI
688 spiSend(token);
689 for (uint16_t i = 0; i < 512; i++) {
690 spiSend(src[i]);
691 }
692 #endif // OPTIMIZE_HARDWARE_SPI
693 spiSend(0xff); // dummy crc
694 spiSend(0xff); // dummy crc
695
696 status_ = spiRec();
697 if ((status_ & DATA_RES_MASK) != DATA_RES_ACCEPTED) {
698 error(SD_CARD_ERROR_WRITE);
699 chipSelectHigh();
700 return false;
701 }
702 return true;
703}
