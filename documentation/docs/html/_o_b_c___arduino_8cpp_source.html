106float OBC_Arduino::thermistor_voltageConversion(float Vtemp)
107{
108 /* Parameters (!!! need to be updated if changements in the circuit) */
109 float_t Vbias = 5.0; // Biasing Tension ! unaccurate
110 float_t Rbias = 10000; // Biasing Resistor
111
112
113 float_t Thermistor_T = 0; // Temperature converted in Celsius degrees
114 float_t R = 0; // Resistor Value
115
116 // Parameters in Python Program
117 float_t A = 1.125966e-3;
118 float_t B = 2.3457279e-4;
119 float_t C = 8.6463018e-8;
120
121 R = Rbias*(Vbias/Vtemp-1); // Thermistor value calculation from voltage
122 Thermistor_T = 1/(A+B*log(R)+C*pow(log(R),3))-273.15; // Steinhart-Hart law
123 return Thermistor_T;
124}
