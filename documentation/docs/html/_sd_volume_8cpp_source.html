30// find a contiguous group of clusters
31uint8_t SdVolume::allocContiguous(uint32_t count, uint32_t* curCluster) {
32 // start of group
33 uint32_t bgnCluster;
34
35 // flag to save place to start next search
36 uint8_t setStart;
37
38 // set search start cluster
39 if (*curCluster) {
40 // try to make file contiguous
41 bgnCluster = *curCluster + 1;
42
43 // don't save new start location
44 setStart = false;
45 } else {
46 // start at likely place for free cluster
47 bgnCluster = allocSearchStart_;
48
49 // save next search start if one cluster
50 setStart = 1 == count;
51 }
52 // end of group
53 uint32_t endCluster = bgnCluster;
54
55 // last cluster of FAT
56 uint32_t fatEnd = clusterCount_ + 1;
57
58 // search the FAT for free clusters
59 for (uint32_t n = 0;; n++, endCluster++) {
60 // can't find space checked all clusters
61 if (n >= clusterCount_) {
62 return false;
63 }
64
65 // past end - start from beginning of FAT
66 if (endCluster > fatEnd) {
67 bgnCluster = endCluster = 2;
68 }
69 uint32_t f;
70 if (!fatGet(endCluster, &f)) {
71 return false;
72 }
73
74 if (f != 0) {
75 // cluster in use try next cluster as bgnCluster
76 bgnCluster = endCluster + 1;
77 } else if ((endCluster - bgnCluster + 1) == count) {
78 // done - found space
79 break;
80 }
81 }
82 // mark end of chain
83 if (!fatPutEOC(endCluster)) {
84 return false;
85 }
86
87 // link clusters
88 while (endCluster > bgnCluster) {
89 if (!fatPut(endCluster - 1, endCluster)) {
90 return false;
91 }
92 endCluster--;
93 }
94 if (*curCluster != 0) {
95 // connect chains
96 if (!fatPut(*curCluster, bgnCluster)) {
97 return false;
98 }
99 }
100 // return first cluster number to caller
101 *curCluster = bgnCluster;
102
103 // remember possible next free cluster
104 if (setStart) {
105 allocSearchStart_ = bgnCluster + 1;
106 }
107
108 return true;
109}

111uint8_t SdVolume::cacheFlush(uint8_t blocking) {
112 if (cacheDirty_) {
113 if (!sdCard_->writeBlock(cacheBlockNumber_, cacheBuffer_.data, blocking)) {
114 return false;
115 }
116
117 if (!blocking) {
118 return true;
119 }
120
121 // mirror FAT tables
122 if (!cacheMirrorBlockFlush(blocking)) {
123 return false;
124 }
125 cacheDirty_ = 0;
126 }
127 return true;
128}

133 return false;
134 }
135 cacheMirrorBlock_ = 0;
136 }
137 return true;

140uint8_t SdVolume::cacheRawBlock(uint32_t blockNumber, uint8_t action) {
141 if (cacheBlockNumber_ != blockNumber) {
142 if (!cacheFlush()) {
143 return false;
144 }
145 if (!sdCard_->readBlock(blockNumber, cacheBuffer_.data)) {
146 return false;
147 }
148 cacheBlockNumber_ = blockNumber;
149 }
150 cacheDirty_ |= action;
151 return true;
152}
153//------------------------------------------------------------------------------
154// cache a zero block for blockNumber
155uint8_t SdVolume::cacheZeroBlock(uint32_t blockNumber) {
156 if (!cacheFlush()) {
157 return false;
158 }
159
160 // loop take less flash than memset(cacheBuffer_.data, 0, 512);
161 for (uint16_t i = 0; i < 512; i++) {
162 cacheBuffer_.data[i] = 0;
163 }
164 cacheBlockNumber_ = blockNumber;
165 cacheSetDirty();
166 return true;
167}

170uint8_t SdVolume::chainSize(uint32_t cluster, uint32_t* size) const {
171 uint32_t s = 0;
172 do {
173 if (!fatGet(cluster, &cluster)) {
174 return false;
175 }
176 s += 512UL << clusterSizeShift_;
177 } while (!isEOC(cluster));
178 *size = s;
179 return true;
180}

182// Fetch a FAT entry
183uint8_t SdVolume::fatGet(uint32_t cluster, uint32_t* value) const {
184 if (cluster > (clusterCount_ + 1)) {
185 return false;
186 }
187 uint32_t lba = fatStartBlock_;
188 lba += fatType_ == 16 ? cluster >> 8 : cluster >> 7;
189 if (lba != cacheBlockNumber_) {
190 if (!cacheRawBlock(lba, CACHE_FOR_READ)) {
191 return false;
192 }
193 }
194 if (fatType_ == 16) {
195 *value = cacheBuffer_.fat16[cluster & 0XFF];
196 } else {
197 *value = cacheBuffer_.fat32[cluster & 0X7F] & FAT32MASK;
198 }
199 return true;
200}

202// Store a FAT entry
203uint8_t SdVolume::fatPut(uint32_t cluster, uint32_t value) {
204 // error if reserved cluster
205 if (cluster < 2) {
206 return false;
207 }
208
209 // error if not in FAT
210 if (cluster > (clusterCount_ + 1)) {
211 return false;
212 }
213
214 // calculate block address for entry
215 uint32_t lba = fatStartBlock_;
216 lba += fatType_ == 16 ? cluster >> 8 : cluster >> 7;
217
218 if (lba != cacheBlockNumber_) {
219 if (!cacheRawBlock(lba, CACHE_FOR_READ)) {
220 return false;
221 }
222 }
223 // store entry
224 if (fatType_ == 16) {
225 cacheBuffer_.fat16[cluster & 0XFF] = value;
226 } else {
227 cacheBuffer_.fat32[cluster & 0X7F] = value;
228 }
229 cacheSetDirty();
230
231 // mirror second FAT
232 if (fatCount_ > 1) {
233 cacheMirrorBlock_ = lba + blocksPerFat_;
234 }
235 return true;
236}
237//------------------------------------------------------------------------------
238// free a cluster chain
239uint8_t SdVolume::freeChain(uint32_t cluster) {
240 // clear free cluster location
241 allocSearchStart_ = 2;
242
243 do {
244 uint32_t next;
245 if (!fatGet(cluster, &next)) {
246 return false;
247 }
248
249 // free cluster
250 if (!fatPut(cluster, 0)) {
251 return false;
252 }
253
254 cluster = next;
255 } while (!isEOC(cluster));
256
257 return true;
258}

275uint8_t SdVolume::init(Sd2Card* dev, uint8_t part) {
276 uint32_t volumeStartBlock = 0;
277 sdCard_ = dev;
278 // if part == 0 assume super floppy with FAT boot sector in block zero
279 // if part > 0 assume mbr volume with partition table
280 if (part) {
281 if (part > 4) {
282 return false;
283 }
284 if (!cacheRawBlock(volumeStartBlock, CACHE_FOR_READ)) {
285 return false;
286 }
287 part_t* p = &cacheBuffer_.mbr.part[part - 1];
288 if ((p->boot & 0X7F) != 0 ||
289 p->totalSectors < 100 ||
290 p->firstSector == 0) {
291 // not a valid partition
292 return false;
293 }
294 volumeStartBlock = p->firstSector;
295 }
296 if (!cacheRawBlock(volumeStartBlock, CACHE_FOR_READ)) {
297 return false;
298 }
299 bpb_t* bpb = &cacheBuffer_.fbs.bpb;
300 if (bpb->bytesPerSector != 512 ||
301 bpb->fatCount == 0 ||
302 bpb->reservedSectorCount == 0 ||
303 bpb->sectorsPerCluster == 0) {
304 // not valid FAT volume
305 return false;
306 }
307 fatCount_ = bpb->fatCount;
308 blocksPerCluster_ = bpb->sectorsPerCluster;
309
310 // determine shift that is same as multiply by blocksPerCluster_
311 clusterSizeShift_ = 0;
312 while (blocksPerCluster_ != (1 << clusterSizeShift_)) {
313 // error if not power of 2
314 if (clusterSizeShift_++ > 7) {
315 return false;
316 }
317 }
318 blocksPerFat_ = bpb->sectorsPerFat16 ?
319 bpb->sectorsPerFat16 : bpb->sectorsPerFat32;
320
321 fatStartBlock_ = volumeStartBlock + bpb->reservedSectorCount;
322
323 // count for FAT16 zero for FAT32
324 rootDirEntryCount_ = bpb->rootDirEntryCount;
325
326 // directory start for FAT16 dataStart for FAT32
327 rootDirStart_ = fatStartBlock_ + bpb->fatCount * blocksPerFat_;
328
329 // data start for FAT16 and FAT32
330 dataStartBlock_ = rootDirStart_ + ((32 * bpb->rootDirEntryCount + 511) / 512);
331
332 // total blocks for FAT16 or FAT32
333 uint32_t totalBlocks = bpb->totalSectors16 ?
334 bpb->totalSectors16 : bpb->totalSectors32;
335 // total data blocks
336 clusterCount_ = totalBlocks - (dataStartBlock_ - volumeStartBlock);
337
338 // divide by cluster size to get cluster count
339 clusterCount_ >>= clusterSizeShift_;
340
341 // FAT type is determined by cluster count
342 if (clusterCount_ < 4085) {
343 fatType_ = 12;
344 } else if (clusterCount_ < 65525) {
345 fatType_ = 16;
346 } else {
347 rootDirStart_ = bpb->fat32RootCluster;
348 fatType_ = 32;
349 }
350 return true;
351}
