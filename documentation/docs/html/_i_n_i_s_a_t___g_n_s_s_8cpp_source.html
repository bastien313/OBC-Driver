148int32_t INISAT_GNSS::fill_NMEA_buffer(uint32_t timeOutMs)
149{
150 int32_t sizeAvailable = 0;

152
153 while(sizeAvailable == 0){
154 delay(100);
155 if( (millis() - inputTime) > timeOutMs){

160 }
161
162 //Buffer overflow protection.
163 if(sizeAvailable >= DATA_BUFFER_SIZE){

165 }
166
167 if(sizeAvailable > 0){

169 uint8_t *buffPtr = NMEA_Buffer;
170 while(byteRemaining){
171 if(byteRemaining > 255){
172 i2cRead(DataStream_addr, buffPtr, 255);
173 byteRemaining -= 255;
174 buffPtr += 255;
175 }else{
176 i2cRead(DataStream_addr, buffPtr, byteRemaining);
177 byteRemaining -= byteRemaining;
178 }
179 }
180 NMEA_Buffer[sizeAvailable] = 0;
181 }
